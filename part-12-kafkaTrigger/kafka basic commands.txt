
Kafka basic commands:

start:
./zookeeper-server-start.sh  "C:\confluentplatform\etc\kafka\zookeeper.properties"
./kafka-server-start.sh  "C:\confluentplatform\etc\kafka\server.properties"
./schema-registry-start.sh  "C:\confluentplatform\etc\schema-registry\schema-registry.properties"

manage topics:
./kafka-topics.sh  --delete --topic testTopic --zookeeper zookeeper:2181
./kafka-topics.sh  --create --topic testTopic --zookeeper zookeeper:2181 --partitions 2 --replication-factor 1
./kafka-topics.sh  --describe --topic testTopic --zookeeper zookeeper:2181
./kafka-topics.sh  --list --zookeeper zookeeper:2181

produce data:
./kafka-console-producer.sh  –broker-list localhost:9092 –topic testTopic

consume data:
./kafka-console-consumer.sh --bootstrap-server localhost:9092 --topic testTopic --group testGroup
./kafka-console-consumer.sh  --new-consumer --bootstrap-server localhost:9092 --topic testTopic --from-beginning

//stream files
kafka-console-producer.sh --broker-list localhost:9092 --topic my_topic < my_file.txt
